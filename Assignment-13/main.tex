%Question
\section{Question}
\begin{frame}{Question}
    \begin{block}{Probability, Random Variables and Stochastic Processes Chapter 2, Problem 7-11} 
        The number of daily accidents is a Poisson random variable $\textbf{n}$ with parameter $a$. The probability that a single accident is fatal equals $p$. Show that the number $\textbf{m}$ of fatal accidents in one day is a Poisson random variable with parameter $ap$.
    \end{block}
\end{frame}

%Solution
\section{Solution}
\begin{frame}{Known equations}
Given, $\textbf{n}$ is a Poisson random variable with parameter $a$. We know that,
\begin{align}
    \pr{\textbf{n}=n} &= \frac{e^{-a}a^n}{n!}
    \\
    \implies E\cbrak{z^{-\textbf{n}}} &= \sum^{\infty}_{n=0}\frac{e^{-a}a^n}{n!}z^{-n}
    \\
    &= e^{-a}e^{az^{-1}}
    \\
    &= e^{-a\brak{1-z^{-1}}}
    \label{eq:eq-1}
\end{align}
If $n$ accidents happen in a day, the probability that $m$ will be fatal,
\begin{align}
    \pr{\textbf{m}=m|\textbf{n}=n} &= \comb{n}{m}p^mq^{n-m}
\end{align}
\end{frame}

\begin{frame}{Solution}
\begin{align}
    E\cbrak{z^{-\textbf{m}}|\textbf{n}=n} &= \sum^{\infty}_{m=0}\comb{n}{m}p^mq^{n-m}z^{-n}
    \\
    &= (pz^{-1}+q)^{n}
\end{align}
We know that,
\begin{align}
    E\cbrak{z^{-\textbf{m}}} &= E\cbrak{E\cbrak{z^{-\textbf{m}}|\textbf{n}}}
    \\
    \implies E\cbrak{z^{-\textbf{m}}} &= E\cbrak{(pz^{-1}+q)^{n}}
    \\
    &= \sum^{\infty}_{n=0}(pz^{-1}+q)^{n}\frac{e^{-a}a^n}{n!}
    \\
    &= e^{a\brak{pz^{-1}+q}}e^{-a}
    \label{eq:eq-2}
\end{align}
\end{frame}

\begin{frame}{Solution}
We know that,
\begin{align}
    q &= 1-p
\end{align}
By substituting this in equation \eqref{eq:eq-2},
\begin{align}
    E\cbrak{z^{-\textbf{m}}} &= e^{-ap\brak{1-z^{-1}}}
\end{align}
Comparing this with equation \eqref{eq:eq-1}, we get that the \textbf{m} is a Poisson random variable with parameter $ap$.
\end{frame}
